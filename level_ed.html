function showCenteredPopup(err1,err2) { var div = document.createElement('div'); div.style.position = 'fixed'; div.style.top = '50vh'; div.style.left = '50vw'; div.style.transform = 'translate(-50%, -50%)'; div.style.width = '400px'; div.style.height = '200px'; div.style.background = 'white'; div.style.color = 'black'; div.style.padding = '20px'; div.style.borderRadius = '10px'; div.style.boxShadow = '0 0 10px 0px black'; div.style.textAlign = 'center'; div.style.zIndex = '9999'; div.style.display = 'block'; div.style.fontSize = '18px'; var h1 = document.createElement('h1'); h1.style.margin = '0'; h1.style.padding = '0 0 10px 0'; h1.style.fontSize = '18px'; h1.innerText = err1+err2; var button = document.createElement('button'); button.style.background = '#00b3cc'; button.style.border = 'none'; button.style.borderRadius = '5px'; button.style.padding = '10px 20px'; button.style.margin = '10px auto'; button.style.cursor = 'pointer'; button.addEventListener('click', function() { div.parentNode.removeChild(div); }); button.textContent = 'OK'; div.appendChild(h1); div.appendChild(button); document.body.appendChild(div); }
